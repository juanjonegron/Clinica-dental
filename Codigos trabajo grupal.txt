Para comenzar debemos habilitar la característica de virtualización en la UEFI o BIOS si nuestro PC es maquina más antigua y luego descargar e instalar DOCKER 

docker run -d --name trabajo-grupal -p 80:80 nginx  esto crear contenedor llamado trabajo-grupal usando una imagen NGINX como servidor web en el puerto 80 y lo apunta al localhost puerto 80.
Como no existía ni contenedor ni imagen anterior arrojo el mensaje: “Unable to find image 'nginx:latest' locally”
latest: Pulling from library/nginx / el Sistema DOCKER procede a descargar la última versión de NGINX 
Status: Downloaded newer image for nginx:latest / finalize la descarga de NGINX
docker ps --filter \"name=trabajo-grupal\ (opcional) sirve para buscar y filtrar los contenedores que se llaman “trabajo grupal”.

version: '3.8'

services:
  db:
    image: mariadb:latest
    container_name: centro-dental-db
    restart: always
    environment:
      MYSQL_ROOT_PASSWORD: root_password
      MYSQL_DATABASE: centro_dental
      MYSQL_USER: dental_user
      MYSQL_PASSWORD: dental_pass
    ports:
      - "3306:3306"
    volumes:
      - ./database:/docker-entrypoint-initdb.d
      - db_data:/var/lib/mysql
    networks:
      - dental-network

  web:
    build: .
    container_name: trabajo-grupal
    restart: always
    ports:
      - "80:80"
    depends_on:
      - db
    environment:
      DB_HOST: db
      DB_NAME: centro_dental
      DB_USER: dental_user
      DB_PASSWORD: dental_pass
    networks:
      - dental-network

volumes:
  db_data:

networks:
  dental-network:
    driver: bridge
